       ```typescript
       import Decimal from 'decimal.js'
       import { endOfMonth } from 'date-fns'
       import type { ForecastInput, ForecastMonth, BillOccurrence } from './types'
       import { getBillOccurrences } from './bills'
       ```

       `forecastMonths(input: ForecastInput, n: number): ForecastMonth[]`:

       Algorithm:
       ```
       results = []
       cumulativeBalance = new Decimal(input.startingBalancePence)
       currentYear = input.startYear
       currentMonth = input.startMonth

       for i = 0 to n-1:
         monthStart = new Date(currentYear, currentMonth - 1, 1)
         monthEnd = endOfMonth(monthStart)

         // Collect bill occurrences for this month
         billsDue: BillOccurrence[] = []
         totalBillsPence = new Decimal(0)
         for each bill in input.bills:
           occurrences = getBillOccurrences(bill, monthStart, monthEnd)
           for each occurrence:
             billsDue.push({ billId: bill.id, name: bill.name, amountPence: bill.amountPence, dueDate: occurrence, potId: bill.potId })
             totalBillsPence = totalBillsPence.plus(bill.amountPence)

         potAllocationsPence = input.pots.reduce((acc, p) => acc.plus(p.allocatedPence), new Decimal(0))
         disposableIncomePence = new Decimal(input.monthlyIncomePence)
           .minus(potAllocationsPence)
           .minus(totalBillsPence)

         cumulativeBalance = cumulativeBalance.plus(disposableIncomePence)

         results.push({
           year: currentYear,
           month: currentMonth,
           incomePence: input.monthlyIncomePence,
           billsDue,
           totalBillsPence: totalBillsPence.toNumber(),
           potAllocationsPence: potAllocationsPence.toNumber(),
           disposableIncomePence: disposableIncomePence.toNumber(),
           cumulativeBalancePence: cumulativeBalance.toNumber(),
         })

         // Advance month
         currentMonth++
         if currentMonth > 12:
           currentMonth = 1
           currentYear++

       return results
       ```
